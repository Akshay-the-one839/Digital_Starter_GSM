#int_AD
void adc_isr() 
{
   int16 adc_sample=read_adc(ADC_READ_ONLY);
   if(channel<=2&&adc_sample>adc_values[channel])
   {
      adc_values[channel]=adc_sample;
      } else if(channel>2&&adc_sample<adc_values[channel]){
      adc_values[channel]=adc_sample;
   }
   //Zero-cross detection(only for voltage channels)
   if(channel<=2&&(last_val[channel]>THRESHOLD)&&(adc_sample<=THRESHOLD))
   {
      if(!crossed[channel])
      {
         crossed[channel]=1;
         if(seq_index<3)
            sequence[seq_index++]=channel;
      }
   }
   if(channel<=2)
      last_val[channel]=adc_sample;
   //Next logical channel
   if(wait==0)
   {
      switch(channel)
      {
         case 0: channel=1;break;
         case 1: channel=2;break;
         case 2: channel=0;break;
      }
      } else {
      switch(channel)
      {
         case 0: channel=1;break;
         case 1: channel=2;break;
         case 2: channel=3;break;
         case 3: channel=4;break;
         case 4: channel=5;break;
         case 5: channel=0;break;
      }
   }
   //Map logical channel 5 to actual AN9
   
   if(channel==5)
      set_adc_channel(14);//AN9=RB3
      else
      set_adc_channel(channel);//AN0 AN4
      
      read_adc(ADC_START_ONLY);
}

void timer1_isr() 
{
   sec_counter++;
  // if(cyclic_state == 1) {
   if(h_time != 0 || m_time != 0 || s_time != 0) {
      time_en++;

      if(time_en >= 20) {   // approx 1 second (20 ticks as you used)
         time_en = 0;

         if(s_time > 0) {
         
            s_time--;
             
         } else {
            s_time = 59;

            if(m_time > 0) {
               m_time--;
            } else {
               m_time = 59;

               if(h_time > 0) {
                  h_time--;
               } else {
                  // reached 00:00:00 - stay at zero, don't go negative
                  time_h = 0;
                  time_m = 0;
                  time_s = 0;
               }
            }
         }
      }
   }
   
   
   time_en2++;
      if(time_en2 >= 20) {   // approx 1 second (20 ticks as you used)
         time_en2 = 0;

         if(dly < 59) {
            dly++;
         }
      }
   
   if(dry_s!=0){
                 dry_s++;}
   
   if(auto_state==0&&aut==1)
      auto_start_s++;
   
   if(dry_reset_s!=0)
      dry_reset_s++;
      
  
   
   if(sec_counter>=10*5)
   {
      //~5s interval
      update_flag=1;
      sec_counter=0;
   }
   if(dry_s>drys*10)
   {
      amps_level=1;

         auto_start_s=0;
        
      }
   }
   if(dry_reset_s>90*15)
   {
      amps_level=0;
   }
   
   set_timer1(34286);//Reload for ~1s tick

// Add this helper function to auto-advance menu
void advance_to_next_menu(void) {
   menu_number++;
   if (menu_number > MAX_MENU_INDEX) {
      menu_number = 1;  // wrap to first menu
   }
   in_menu_selection = 1;  // Return to menu selection mode
}

// Your existing functions remain the same
void show_menu_selection(void) {
   static int8 last_displayed_menu = 255;  
   
   if (menu_number != last_displayed_menu) {
      lcd_cmd(LCD_CLEAR);   
      
      lcd_goto(1, 1);
      printf(lcd_out, "%s", menu_titles[menu_number]);
      
       lcd_goto(1, 2);
      if (menu_number == MAX_MENU_INDEX) {
         printf(lcd_out, "Press to EXIT");
      } else {
         printf(lcd_out, "Press to ENTER");
      }
      
      last_displayed_menu = menu_number;
   }
}

void handle_menu_navigation(void) {
   while (in_menu_selection) {
      show_menu_selection();
      
      if (input(ENCODER_A) == 0 && input(ENCODER_B) != 0) {
         delay_ms(5);
         menu_number++;
         if (menu_number > MAX_MENU_INDEX) menu_number = 1;
         while (input(ENCODER_A) == 0);
      }
      else if (input(ENCODER_B) == 0 && input(ENCODER_A) != 0) {
         delay_ms(5);
         if (menu_number == 1) 
            menu_number = MAX_MENU_INDEX;
         else 
            menu_number--;
         while (input(ENCODER_B) == 0);
      }
      
      if (input(ENCODER_SW) == 0) {
         delay_ms(20);
         if (input(ENCODER_SW) == 0) {
            while (input(ENCODER_SW) == 0) {
               delay_ms(10);
            }
            
            if (menu_number == MAX_MENU_INDEX) {
               prog = 0;
               in_menu_selection = 0;
               lcd_cmd(LCD_CLEAR);  
               lcd_goto(1, 1);  
               printf(lcd_out, "EXITING MENU...");
               delay_ms(1000);
                lcd_cmd(LCD_CLEAR); 
               return;
            } else {
               in_menu_selection = 0;
               return;
            }
         }
      }
      
      delay_ms(1);
   }
}

void handle_phase_menu(void) {
   // Clear any false press at entry
   while (input(ENCODER_SW) == 0);
    lcd_cmd(LCD_CLEAR);  
    lcd_goto(1, 1);  
   printf(lcd_out, "1.PHASE SETTING");
   
   int8 selected_phase = read_eeprom(0x00);  // 0 = 2 phase, 1 = 3 phase
   int8 original_phase = selected_phase;
   int8 last_displayed = 255;
   
   // Display saved value initially
      
    lcd_goto(1, 2);  
   if (selected_phase == 0)
      printf(lcd_out, ">> 2 PHASE        ");
   else
      printf(lcd_out, ">> 3 PHASE        ");
   
   while (1) {
      // === Encoder Rotation ===
      if (input(ENCODER_A) == 0 && input(ENCODER_B) != 0) {
         delay_ms(5);
         selected_phase = (selected_phase + 1) % 2;
         while (input(ENCODER_A) == 0);
      }
      else if (input(ENCODER_B) == 0 && input(ENCODER_A) != 0) {
         delay_ms(5);
         selected_phase = (selected_phase == 0) ? 1 : 0;
         while (input(ENCODER_B) == 0);
      }
      
      // === Update LCD only if changed ===
      if (selected_phase != last_displayed) {
         lcd_goto(1, 2); 
         if (selected_phase == 0)
            printf(lcd_out, ">> 2 PHASE        ");
         else
            printf(lcd_out, ">> 3 PHASE        ");
         last_displayed = selected_phase;
      }
      
      // === Handle Button Press ===
      if (input(ENCODER_SW) == 0) {
         delay_ms(20);  // debounce
         if (input(ENCODER_SW) == 0) {
            if (selected_phase != original_phase) {
               // Save new value
               write_eeprom(0x00, selected_phase);
               // Show save confirmation
                 lcd_cmd(LCD_CLEAR);  
                lcd_goto(1, 1);  
                printf(lcd_out, "PHASE SAVED");
                lcd_goto(1, 2);  
               if (selected_phase == 0)
               {
                  printf(lcd_out, "2 PHASE");
                  phase=selected_phase;
               }
               else{
                  printf(lcd_OUT, "3 PHASE");
                  phase=selected_phase;
               }
               // Wait for release and delay
               while (input(ENCODER_SW) == 0);
               delay_ms(1000);
               phase_setting =selected_phase;
               // Auto advance to next menu after saving
               advance_to_next_menu();
            }
            else {
               // No change - still advance to next menu
               while (input(ENCODER_SW) == 0);
               advance_to_next_menu();
            }
            break;  // Exit the while loop
         }
      }
      delay_ms(1);
   }
}

   else if (increment_mode == 10) increment_mode = 100;
   else increment_mode = 1;

   int8 timeout = 50;  // 500ms max wait
   while (input(key1) == 0 && timeout > 0) {
      delay_ms(10);
      timeout--;
   }
}
      if (input(key2) == 0) {
   delay_ms(50);

         }
         delay_ms(40);
         while (input(ENCODER_B) == 0);  // Wait for release

            printf(lcd_out, "%s SUF", title);

         lcd_goto(1, 2); 
         printf(lcd_out, ">> %ld", (long)value);  // Use %ld with cast
         
         last_value = value;
         last_mode = increment_mode;
      }

      if (!input(ENCODER_SW)) {
         delay_ms(50);
         while (!input(ENCODER_SW));  // wait release
         break; // Exit and return value
         }
          delay_ms(5);
   }

void  handle_cyclic_timer(void) {
   // Clear any false press at entry
   while (input(ENCODER_SW) == 0);  // Wait if button is already pressed
   
   // Load saved values from EEPROM
   int8 on_hr = read_eeprom(0x20);
   int8 on_min = read_eeprom(0x21);
   int8 off_hr = read_eeprom(0x22);
   int8 off_min = read_eeprom(0x23);
   
   // Validate ranges
   if (on_hr > 23) on_hr = 0;
   if (on_min > 59) on_min = 0;
   if (off_hr > 23) off_hr = 0;
   if (off_min > 59) off_min = 0;
   
   // Store original values for comparison
   int8 original_on_hr = on_hr;
   int8 original_on_min = on_min;
   int8 original_off_hr = off_hr;
   int8 original_off_min = off_min;
   
   int8 field = 0; // 0=on_hr, 1=on_min, 2=off_hr, 3=off_min
   int8 last_field = -1;
   int8 last_on_hr = -1, last_on_min = -1, last_off_hr = -1, last_off_min = -1;
   int1 button_pressed_flag = 0;  // Flag to track button press state
   
   lcd_cmd(LCD_CLEAR); 
   delay_ms(100);  // Allow LCD to fully clear
   lcd_cmd(LCD_CLEAR);                    // Clear the LCD
   lcd_goto(1, 1); 
   printf(lcd_out, "8.CYCLIC TIMER");
   delay_ms(1000);
   
   while (1) {
      // Update display only if values changed
      if (field != last_field || on_hr != last_on_hr || on_min != last_on_min || 
          off_hr != last_off_hr || off_min != last_off_min) {
         
         lcd_goto(1, 1); 
         printf(lcd_out, "ON :");
         if (field == 0) printf(lcd_out, "[%02u]", on_hr);
         else            printf(lcd_out, " %02u ", on_hr);
         printf(lcd_out, ":");
         if (field == 1) printf(lcd_out, "[%02u]  ", on_min);  // Extra padding
         else            printf(lcd_out, " %02u   ", on_min);   // Extra padding
         
        lcd_goto(1, 2); 
         printf(lcd_out, "OFF:");
         if (field == 2) printf(lcd_out, "[%02u]", off_hr);
         else            printf(lcd_out, " %02u ", off_hr);
         printf(lcd_out, ":");
         if (field == 3) printf(lcd_out, "[%02u]   ", off_min); // Extra padding
         else            printf(lcd_out, " %02u    ", off_min);  // Extra padding
         
         // Update last values
         last_field = field;
         last_on_hr = on_hr;
         last_on_min = on_min;
         last_off_hr = off_hr;
         last_off_min = off_min;
      }
      
      // === Encoder rotation logic ===
      // Wait for both pins to be high (idle state)
      if (input(ENCODER_A) && input(ENCODER_B)) {
         // Now wait for one pin to go low (start of rotation)
         while (input(ENCODER_A) && input(ENCODER_B)) {
            // Check for other button presses while waiting
            if (input(KEY1) == 0 || input(ENCODER_SW) == 0) break;
         }
         
         // Check which pin went low first to determine direction
         if (input(ENCODER_A) == 0 && input(ENCODER_B) == 1) {
            delay_ms(5); // Debounce
            if (input(ENCODER_A) == 0 && input(ENCODER_B) == 1) { // Confirm state
               switch (field) {
                  case 0: if (on_hr < 23) on_hr++; break;
                  case 1: if (on_min < 59) on_min++; break;
                  case 2: if (off_hr < 23) off_hr++; break;
                  case 3: if (off_min < 59) off_min++; break;
               }
               // Wait for encoder to return to idle
               while (input(ENCODER_A) == 0 || input(ENCODER_B) == 0);
               delay_ms(50); // Prevent multiple triggers
            }
         }
         else if (input(ENCODER_B) == 0 && input(ENCODER_A) == 1) {
            delay_ms(5); // Debounce
            if (input(ENCODER_B) == 0 && input(ENCODER_A) == 1) { // Confirm state
               switch (field) {
                  case 0: if (on_hr > 0) on_hr--; break;
                  case 1: if (on_min > 0) on_min--; break;
                  case 2: if (off_hr > 0) off_hr--; break;
                  case 3: if (off_min > 0) off_min--; break;
               }
               // Wait for encoder to return to idle
               while (input(ENCODER_A) == 0 || input(ENCODER_B) == 0);
               delay_ms(50); // Prevent multiple triggers
            }
         }
      }
      
      // === Switch field on KEY1 press ===
      if (input(KEY1) == 0) {
         delay_ms(50);
     
         field = (field + 1) % 4;
          
         int8 timeout = 50;  // 500ms max wait
         while (input(key1) == 0 && timeout > 0) {
         delay_ms(10);
         timeout--;
   }
 }
   
     // === Handle Encoder Switch Press ===
      if (input(ENCODER_SW) == 0 && button_pressed_flag == 0) {
         delay_ms(20);  // debounce
         if (input(ENCODER_SW) == 0) {  // confirm still pressed
            button_pressed_flag = 1;
            
            // Check if any values were actually changed
            if (on_hr != original_on_hr || on_min != original_on_min || 
                off_hr != original_off_hr || off_min != original_off_min) {
               // Values changed - save and show confirmation immediately
               
               // Save to EEPROM
               write_eeprom(0x20, on_hr);
               write_eeprom(0x21, on_min);
               write_eeprom(0x22, off_hr);
               write_eeprom(0x23, off_min);
               
               // Show confirmation immediately
               lcd_cmd(LCD_CLEAR); 
               delay_ms(50);   // Allow LCD to clear
                lcd_goto(1, 1); 
               printf(lcd_out, "TIMER SAVED!    ");  // Extra padding to clear any garbage
            
               
               // Wait for button release while showing message
               while (input(ENCODER_SW) == 0) {
                  delay_ms(10);
               }
               delay_ms(1500);  
                advance_to_next_menu();
            }
            else {
               // No change - wait for release and go directly to next menu
               while (input(ENCODER_SW) == 0) {
                  delay_ms(10);
               }
                advance_to_next_menu();
            }
            
            // Proper LCD reset before exit
            lcd_cmd(LCD_CLEAR); 
            delay_ms(100);
            
            // Exit the while loop
            break;
         }
         else {
            // False press - reset flag
            button_pressed_flag = 0;
         }
      }
      
      // Reset button flag when button is released
      if (input(ENCODER_SW) != 0) {
         button_pressed_flag = 0;
      }
      
      delay_ms(1);  // Small delay for stability
   }
}

void handle_auto_restart_mode() {
  
    // Clear any false encoder press at entry
   while (input(ENCODER_SW) == 0);
   lcd_cmd(LCD_CLEAR); 
   lcd_goto(1, 1);
   printf(lcd_out, "9.AUTO RESTART");
   
   int8 auto_restart = read_eeprom(0x24);  // 0 = Auto, 1 = Manual
   int8 original_mode = auto_restart;      // Store original value for comparison
   int8 last_displayed = 255;
   int1 button_pressed_flag = 0; 
    
    // Display saved value initially
   lcd_goto(1, 2);
   if (auto_restart == 0)
      printf(lcd_out, ">> MODE: AUTO     ");
   else
      printf(lcd_out, ">> MODE: MANUAL   ");
   
   while (1) {
      // === Encoder Rotation ===
      if (input(ENCODER_A) == 0 && input(ENCODER_B) != 0) {
         delay_ms(5);
         auto_restart++;
         if (auto_restart > 1) auto_restart = 0;
         while (input(ENCODER_A) == 0); // wait release
      }

         delay_ms(5);
         if (auto_restart == 0)
            auto_restart = 1;
         else
            auto_restart--;
         while (input(ENCODER_B) == 0); // wait release
      }

      if (auto_restart != last_displayed) {
         
         lcd_goto(1, 2);
         if (auto_restart == 0)
            printf(lcd_out, ">> MODE: AUTO     ");
         else
            printf(lcd_out, ">> MODE: MANUAL   ");

      // === Handle Encoder Switch Press ===
      if (input(ENCODER_SW) == 0 && button_pressed_flag == 0) {
         delay_ms(20);  // debounce
         if (input(ENCODER_SW) == 0) {  // confirm still pressed
            button_pressed_flag = 1;
            
            // Check if value was actually changed

               // Value changed - save and show confirmation immediately

              lcd_cmd(LCD_CLEAR); 
               lcd_goto(1, 1);
               printf(lcd_out, "MODE SAVED!");

                  printf(lcd_out, "MODE: MANUAL ");
               
               // Wait for button release while showing message
               while (input(ENCODER_SW) == 0) {
                  delay_ms(10);
               }
               delay_ms(1000);  
                advance_to_next_menu();
            }
            else {
               // No change - wait for release and go directly to next menu
               while (input(ENCODER_SW) == 0) {
                  delay_ms(10);
               }
                advance_to_next_menu();
            }
            
            // Proper LCD reset before exit
           lcd_cmd(LCD_CLEAR);     
            delay_ms(100);
            
            // Exit the while loop
            break;
         }
         else {
            // False press - reset flag
            button_pressed_flag = 0;
         }
      }
      
      // Reset button flag when button is released
      if (input(ENCODER_SW) != 0) {
         button_pressed_flag = 0;
      }
      
      delay_ms(1);  // Smooth loop delay
   }
}

void handle_ct_selection_menu(void) {
    // Clear any false press at entry
    while (input(ENCODER_SW) == 0);  // Wait if button is already pressed
    
    // CT primary amps and corresponding ratio (user view vs calculation)
    const int16 ct_amps[]  = {5, 30, 50, 100, 200, 300};
    const int8  ct_ratios[] = {1, 6, 10, 20, 40, 60};
    const int8 total_cts = sizeof(ct_amps) / sizeof(ct_amps[0]);
   
    // Load saved CT ratio from EEPROM and find corresponding option
    int8 saved_ratio = read_eeprom(0x60);
    int8 original_ratio = saved_ratio;  // Store original value for comparison
    int8 option = 0;  // Default to first option
    
    // Find the option that matches the saved ratio
    for (int8 i = 0; i < total_cts; i++) {
        if (ct_ratios[i] == saved_ratio) {
            option = i;
            break;
        }
    }
    
    // If saved ratio is invalid, default to first option and save it
    if (saved_ratio != ct_ratios[option]) {
        write_eeprom(0x60, ct_ratios[0]);  // Save default
        original_ratio = ct_ratios[0];
        option = 0;
    }

        if (input(ENCODER_A) == 0 && input(ENCODER_B) != 0) {

        }
        else if (input(ENCODER_B) == 0 && input(ENCODER_A) != 0) {
            delay_ms(5);
            
            if (option > 0) option--;  // Don't wrap around at beginning
            delay_ms(40);
            while (input(ENCODER_B) == 0);  // Prevent rapid changes
        }
        
        // === Handle Encoder Switch Press ===
        if (input(ENCODER_SW) == 0) {
            delay_ms(20);  // debounce
            if (input(ENCODER_SW) == 0) {  // confirm still pressed
                
                // Check if CT selection has changed
                if (ct_ratios[option] != original_ratio) {
                    // CT CHANGED: Save to EEPROM and show confirmation
                    write_eeprom(0x60, ct_ratios[option]);  // Save new CT ratio to EEPROM
                    
                     lcd_cmd(LCD_CLEAR); 
                     lcd_goto(1, 1);

                    lcd_goto(1, 2);
                    printf(lcd_out, "CT: %luA (%u:1)", (unsigned long)ct_amps[option], ct_ratios[option]);
                    
                    // Wait for button release while showing message
                    while (input(ENCODER_SW) == 0) {
                        delay_ms(10);
                    }
                    delay_ms(1000);  // Show message for 1 second after release
                    
                    // Auto advance to next menu after saving
                    advance_to_next_menu();
                }
                else {
                    // No change - wait for release and auto advance to next menu directly
                    while (input(ENCODER_SW) == 0) {
                        delay_ms(10);
                    }
                    
                    // Auto advance to next menu without showing save message
                    advance_to_next_menu();
                }
                
                // Proper LCD reset before exit
                lcd_cmd(LCD_CLEAR); 
                   
                delay_ms(100);
                
                // Exit the while loop
                break;
            }
        }
        
        delay_ms(1);  // Small delay for stability
    }
}
void handle_polarization_index_menu(void) {
    // Clear any false press at entry

    // Load saved values from EEPROM or use defaults
    int16 ir_1min_int = make16(read_eeprom(0x31), read_eeprom(0x30));  // Store as 10x value (5 = 0.5)
    int16 ir_10min_int = make16(read_eeprom(0x33), read_eeprom(0x32)); // Store as 10x value (10 = 1.0)

    if (ir_1min_int < 5 || ir_1min_int > 1000) ir_1min_int = 5;    // Default 0.5 MOhm
    if (ir_10min_int < 5 || ir_10min_int > 1000) ir_10min_int = 10; // Default 1.0 MOhm
    
    // Store original values for comparison
    int16 original_ir_1min = ir_1min_int;
    int16 original_ir_10min = ir_10min_int;
    
    float ir_1min, ir_10min, pii;
    int16 last_value = -1;
    int1 button_pressed_flag = 0;  // Flag to track button press state
    int1 values_changed = 0;       // Track if any values changed
    
     lcd_cmd(LCD_CLEAR); 
                   
    delay_ms(100);
    lcd_cmd(LCD_CLEAR); 
    lcd_goto(1, 1);
    printf(lcd_out, "12.PI TEST-IR@1m");
   
    
    // Step 1: IR at 1min
    while (1) {
        ir_1min = (float)ir_1min_int / 10.0;  // Convert back to float
        
        // Show value only if changed
        if (ir_1min_int != last_value) {
            lcd_goto(1, 2);
            printf(lcd_out, " : %3.1f MOhm   ", ir_1min);
            last_value = ir_1min_int;
        }
        
        // Detect falling edge on A with B high (clockwise - increment)
        if (input(ENCODER_A) == 0 && input(ENCODER_B) != 0) {
            delay_ms(5);  // Debounce
            
            if (ir_1min_int < 1000) ir_1min_int += 5;  // Increment by 0.5
            delay_ms(40);
            while (input(ENCODER_A) == 0);  // Prevent rapid changes
        }
        else if (input(ENCODER_B) == 0 && input(ENCODER_A) != 0) {
            delay_ms(5);
            
            if (ir_1min_int > 5) ir_1min_int -= 5;  // Decrement by 0.5
            delay_ms(40);
            while (input(ENCODER_B) == 0);  // Prevent rapid changes
        }
        
        // === Enhanced Save Logic for Step 1 ===
        if (input(ENCODER_SW) == 0) {
            delay_ms(50);
            button_pressed_flag = 1;
            
            // Check if IR@1min value has changed
            if (ir_1min_int != original_ir_1min) {
                values_changed = 1;

                lcd_cmd(LCD_CLEAR); 
                lcd_goto(1, 1);
                printf(lcd_out, "IR@1m SAVED!");
              lcd_goto(1, 2);
                printf(lcd_out, "Value: %3.1f MOhm", ir_1min);
                
                // Wait for button release while showing confirmation
                while (input(ENCODER_SW) == 0){
                   delay_ms(10);
                }
                // Show confirmation for 1 more second after release
                delay_ms(1000);
            }
            else {
                // NO CHANGES: Wait for release then continue
                while (input(ENCODER_SW) == 0){
                    delay_ms(10);
                }
            }
            
            break;  // Move to step 2
        }
        
        delay_ms(1);  // Small delay for stability
    }
    
    // Step 2: IR at 10min
    last_value = -1;
    lcd_cmd(LCD_CLEAR); 
                
    delay_ms(50);
    lcd_goto(0, 1);
    printf(lcd_out, "Enter IR@10min  ");
    
    while (1) {
        ir_10min = (float)ir_10min_int / 10.0;  // Convert back to float
        
        // Show value only if changed
        if (ir_10min_int != last_value) {
            lcd_goto(1, 2);
            printf(lcd_out, " : %3.1f MOhm   ", ir_10min);
            last_value = ir_10min_int;
        }
        
        // Detect falling edge on A with B high (clockwise - increment)
        if (input(ENCODER_A) == 0 && input(ENCODER_B) != 0) {
            delay_ms(5);  // Debounce
            
            if (ir_10min_int < 1000) ir_10min_int += 5;  // Increment by 0.5
            delay_ms(40);
            while (input(ENCODER_A) == 0);  // Prevent rapid changes
        }
        else if (input(ENCODER_B) == 0 && input(ENCODER_A) != 0) {
            delay_ms(5);
            
            if (ir_10min_int > 5) ir_10min_int -= 5;  // Decrement by 0.5
            delay_ms(40);
            while (input(ENCODER_B) == 0);  // Prevent rapid changes
        }
        
        // === Enhanced Save Logic for Step 2 ===
        if (input(ENCODER_SW) == 0) {
            delay_ms(50);
            button_pressed_flag = 1;
            
            // Check if IR@10min value has changed
            if (ir_10min_int != original_ir_10min) {
                values_changed = 1;
            }
            
            // Check if ANY values changed during the entire process
            if (values_changed || (ir_1min_int != original_ir_1min) || (ir_10min_int != original_ir_10min)) {
                // VALUES CHANGED: Show confirmation immediately

                lcd_goto(1, 2);

                while (input(ENCODER_SW) == 0) {
                    delay_ms(10);
                }
                
                // Save values to EEPROM (only when changed)
                write_eeprom(0x30, make8(ir_1min_int, 0));  // LSB of IR@1min
                write_eeprom(0x31, make8(ir_1min_int, 1));  // MSB of IR@1min
                write_eeprom(0x32, make8(ir_10min_int, 0)); // LSB of IR@10min
                write_eeprom(0x33, make8(ir_10min_int, 1)); // MSB of IR@10min
                
                // Show confirmation for 1 more second after release
                delay_ms(1000);
            }
            else {
                // NO CHANGES: Wait for release then continue
                while (input(ENCODER_SW) == 0) {
                    delay_ms(10);
                }
                // No EEPROM write, no confirmation message
            }
            
            break;  // Move to results
        }
        
        delay_ms(1);  // Small delay for stability
    }
    
    // Step 3: Calculate PI and show result
    ir_1min = (float)ir_1min_int / 10.0;
    ir_10min = (float)ir_10min_int / 10.0;
    pii = ir_10min / ir_1min;
    
   lcd_cmd(LCD_CLEAR);      
                
    delay_ms(50);

    printf(lcd_out, "PI=%3.2f RESULT ", pii);
   lcd_goto(1, 2);

        printf(lcd_out, "GOOD INSULATION ");
    else if (pii >= 1.0)
        printf(lcd_out, "POOR INSULATION ");
    else
        printf(lcd_out, "BAD INSULATION  ");
    
    delay_ms(3000);  // Show result for 3 seconds

    advance_to_next_menu();
    
    // Proper LCD reset before exit
   lcd_cmd(LCD_CLEAR); 
    delay_ms(100);
}

void check_unbalance_trip() {
 

   float r = ramp;
   float y = yamp;
   float b = bamp;

   float max = r;
   if (y > max) max = y;
   if (b > max) max = b;

   float min = r;
   if (y < min) min = y;
   if (b < min) min = b;

   float avg = (r + y + b) / 3.0;
   if (avg == 0.0){
   trip_status=0;
   return;  // Avoid divide-by-zero error
   }
   
   float unbalance = ((max - min) / avg) * 100.0;

   // Read threshold from EEPROM
   int16 ub_limit = make16(read_eeprom(0x27), read_eeprom(0x26));
   if (ub_limit > 100) ub_limit = 30;

   if (unbalance > ub_limit) {
      output_low(PIN_C1);  // Motor OFF
        trip_status = 1;     // Set trip
    } else {
        trip_status = 0;     // Clear trip
    }
}
